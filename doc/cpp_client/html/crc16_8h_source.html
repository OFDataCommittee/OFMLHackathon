34 /* CRC16 implementation according to CCITT standards.
35  *
36  * Note by @antirez: this is actually the XMODEM CRC 16 algorithm, using the
37  * following parameters:
38  *
39  * Name : "XMODEM", also known as "ZMODEM", "CRC-16/ACORN"
40  * Width : 16 bit
41  * Poly : 1021 (That is actually x^16 + x^12 + x^5 + 1)
42  * Initialization : 0000
43  * Reflect Input byte : False
44  * Reflect Output CRC : False
45  * Xor constant to output CRC : 0000
46  * Output for "123456789" : 31C3
47  */

84 uint16_t crc16(const char *buf, int len) {
85  int counter;
86  uint16_t crc = 0;
87  for (counter = 0; counter < len; counter++)
88  crc = (crc<<8) ^ crc16tab[((crc>>8) ^ *buf++)&0x00FF];
89  return crc;
90 }
